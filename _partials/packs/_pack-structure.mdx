│   │   ├── templates

│   ├── chart-1.tgz
│   ├── chart-2
│   │   ├── Chart.yaml
│   │   ├── templates
│   │   │   ├── template-1.yaml
│   │   │   └── template-2.yaml
│   │   └── values.yaml # Chart-level values.yaml file.
│   └── chart-2.tgz
├── logo.png            # Mandatory.
└── README.md           # Optional.
